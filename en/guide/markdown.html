                                                  # Theme Enhancement

                                                  # Enable all

                                                  You can set themeconfig.plugins.htmlEnhance.enableAll to enable all features of the md-enhanceopen in new window plugin.

                                                  module.exports = {
  themeConfig: {
    plugins: {
      mdEnhance: {
        enableAll: true,
      },
    },
  },
};
